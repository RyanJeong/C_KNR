void itoa(int, char *);
void reverse(char *);

void itoa(int n, char *buf)
{
    char    *p;
    int     sign;
    
    p       = buf;
    sign    = n;
    while (n) {
        *buf++  = ABS((n % 10)) + '0';
        n       /= 10;
    }
    if (sign < 0) {
        *buf++ = '-';
    }
    *buf = '\0';
    reverse(p);
}

void reverse(char *buf) 
{
    char c, *p;
    
    p = buf;
    while (*p != '\0') {
        ++p;
    }
    --p;    /*  one step before('\0')   */
    while (buf < p) {
        c       = *buf;
        *buf++  = *p;
        *p--    = c;    
    }
}
